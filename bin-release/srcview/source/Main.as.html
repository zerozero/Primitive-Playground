package
{
    
    import com.adobe.viewsource.ViewSource;
    import com.bit101.components.ComboBox;
    import com.bit101.components.PushButton;
    import com.jonrowe.away3d.productFactory.ProductType;
    
    import flash.display.Sprite;
    import flash.events.Event;
    import flash.text.TextField;
    
    [SWF("800", height="600", frameRate="60")]
    public class Main extends Sprite
    {
        private static const PAD :int = 10;
        
        private var combo :ComboBox;
        private var asView3D :ASView3D;
        
        public var onjects :Array = new Array(
            
                {label:ProductType.WIRE_SPHERE},    
                {label:ProductType.WIRE_CUBE},    
                {label:ProductType.WIRE_PLANE},    
                {label:ProductType.CAPSULE},    
                {label:ProductType.CONE}    
            
        );
        
        public function Main()
        {
            asView3D = new ASView3D();
            asView3D.createChildren();
            
            asView3D.x = 200;
            addChild (asView3D);
            
            createGUI();
            
            ViewSource.addMenuItem(this, "srcview/index.html"); 
        }
        
        private function createGUI():void{
            
            var x :uint = 10;
            var y :uint = 10;
            
            var t :TextField = new TextField();
            t.x = x;
            t.y = y;
            t.text = "properties";
            t.height = t.textHeight + 10;
            addChild(t);
            
            y+= t.height + PAD;
            
            combo = new ComboBox(this,x,y,"Select..",onjects);
            combo.setSize( 160, combo.height );
            addChild(combo);
            
            y += combo.height +PAD;
            
            var pb :PushButton = new PushButton(this, x, y, "Make", onMake);
            
        }
        
        private function onMake(e:Event):void{
            var selected :Object = combo.selectedItem;
            asView3D.createObject(selected.label);
        }
    }
}
