package 
{
    import away3d.containers.ObjectContainer3D;
    import away3d.containers.View3D;
    import away3d.core.base.Object3D;
    import away3d.debug.AwayStats;
    import away3d.debug.Trident;
    import away3d.lights.DirectionalLight;
    import away3d.primitives.WireframeAxesGrid;
    import away3d.primitives.WireframeSphere;
    
    import com.jonrowe.away3d.productFactory.ProductFactory;
    import com.jonrowe.away3d.productFactory.ProductType;
    import com.jonrowe.away3d.productFactory.ProductTypeMap;
    import com.jonrowe.away3d.productFactory.interfaces.IProduct;
    import com.jonrowe.away3d.productFactory.onjects.Onject;
    import com.jonrowe.away3d.utils.HoverDragController;
    
    import flash.display.Sprite;
    import flash.events.Event;
    import flash.geom.Vector3D;
    import com.jonrowe.away3d.view.SpriteView3D;
    
    public class ASView3D extends Sprite
    {
        /*        PUBLIC PROPERTIES        */

        
        /*        PUBLIC PROPERTIES        */
        
        /*        PRIVATE PROPERTIES        */
        
        private var _view:View3D;
        private var spriteView3D :SpriteView3D;
        private var camController:HoverDragController;   
        private var directionalLight :DirectionalLight;
        
        /*        PRIVATE PROPERTIES        */
        
        /*        CONSTRUCT        */
        
        public function ASView3D()
        {
            super();
            this.addEventListener( Event.ADDED_TO_STAGE, onAddedToStage );
        }
        
        /*        CONSTRUCT        */
        
        /*        OVERRIDE        */
        
        public function createChildren():void
        {
            
            if (!spriteView3D){
                spriteView3D = new SpriteView3D();
                this._view = spriteView3D.view;
            }
            this.addChild( spriteView3D );
            
            
        }
        
        
        /*override protected function updateDisplayList(
            unscaledWidth:Number, unscaledHeight:Number):void
        {
            
            super.updateDisplayList(unscaledWidth, unscaledHeight);
            
            _view.width = this.width;
            _view.height = this.height;
            
        }*/
        
        /*        OVERRIDE        */
        
        /*        PUBLIC        */
        
        public function createObject( type :String ):IProduct{
            var product :IProduct = Onject.create( type );
            _view.scene.addChild(product.object3D);
            product.light([directionalLight]);
            product.select();
            return product;
        }
        
        /*        PUBLIC        */
        
        /*        PRIVATE        */
        
        private function initView():void{
            _view.width = 600;
            _view.height = 600;
        }
        
        private function initScene():void{
            
            var gridPlane :IProduct = createObject( ProductType.AXES );
            //var trident :IProduct = createObject( ProductType.TRIDENT );
            
            var stats :AwayStats = new AwayStats(_view);
            addChild(stats);
            
            addEventListener(Event.ENTER_FRAME, onEnterFrame);
            
        }
        
        private function initCamController():void{
            camController = new HoverDragController(_view.camera, this.stage);
            camController.radius = 500;
        }
        
        private function initLights():void{
            directionalLight = new DirectionalLight();
            directionalLight.specular = 0.3;
            _view.scene.addChild(directionalLight);
        }
        
        /*        PRIVATE        */
        
        
        /*        EVENT        */
        
        
        private function onAddedToStage( e:Event ):void{
            initView();
            initScene();
            initCamController();
            initLights();
        }
        
        private function onEnterFrame(ev : Event) : void {
            directionalLight.rotate(new Vector3D(1,0,0),2);
            _view.render();
            
        } 

        /*        EVENT        */
        
    }
}
