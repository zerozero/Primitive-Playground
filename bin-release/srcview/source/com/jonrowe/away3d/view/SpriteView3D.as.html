package com.jonrowe.away3d.view
{
    import away3d.containers.View3D;
    
    import flash.display.Sprite;
    
    public class SpriteView3D extends Sprite
    {
        private var _view:View3D;
        
        public function SpriteView3D()
        {
            super();
            if(!this._view)
            {
                _view = new View3D(); 
                _view.backgroundColor = 0x000000; 
                _view.antiAlias = 4; 
            }
            this.addChild(_view); 
        }
        
        public function get view():View3D{
            return _view;
        }
    }
}
